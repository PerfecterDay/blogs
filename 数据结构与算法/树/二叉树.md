### DFS-Depth First Search
```
递归中序遍历
public void traverseInOrder(Node node) {
    if (node != null) {
        traverseInOrder(node.left);
        System.out.print(" " + node.value);
        traverseInOrder(node.right);
    }
}


* 借助栈实现迭代的中序遍历，返回的排序的数列
* */
private void iteInorderTraverse(Node node) {
    if (node == null) {
        return;
    }
    Stack<Node> stack = new Stack();
    while (stack.size() > 0 || node != null) {
        if (node != null){
            stack.push(node);
            node = node.left;
        }else {
            Node curr = stack.pop();
            System.out.println(curr.value);
            node = curr.right;
        }
    }
}

递归前序遍历
public void traversePreOrder(Node node) {
    if (node != null) {
        System.out.print(" " + node.value);
        traversePreOrder(node.left);
        traversePreOrder(node.right);
    }
}

/*
* 借助栈实现迭代的前序遍历，返回的排序的数列
* */
private void itePreorderTraverse(Node node) {
    if (node == null) {
        return;
    }
    Stack<Node> stack = new Stack();
    stack.push(node);
    while (!stack.isEmpty()) {
        Node indexNode = stack.pop();
        System.out.println(indexNode.value);
        if (indexNode.right != null) {
            stack.push(indexNode.right);
        }
        if (indexNode.left != null) {
            stack.push(indexNode.left);
        }
    }
}

后序遍历
public void traversePreOrder(Node node) {
    if (node != null) {
        System.out.print(" " + node.value);
        traversePreOrder(node.left);
        traversePreOrder(node.right);
    }
}


/*
* 借助栈实现迭代的后序遍历，返回的排序的数列
* */
private void itePostOderTraverse(Node node) {
    if (node == null) {
        return;
    }
    Stack<Node> stack = new Stack<>();
    Node prev = null;
    while (node != null || !stack.isEmpty()) {
        while (node != null) {
            stack.push(node);
            if (node.left!=null){
                node = node.left;
            }else {
                node = node.right;
            }
        }
        node = stack.pop();
        if (node.right == null || node.right == prev) {
            System.out.println(node.value);
            prev = node;
            node = null;
        } else {
            stack.push(node);
            node = node.right;
        }
    }
}

### BFS-Breadth First Search
```
public void traverseLevelOrder() {
    if (node == null) {
        return;
    }
    Queue<Node> queue = new LinkedList();
    queue.add(node);
    while (!queue.isEmpty()) {
        Node indexNode = queue.poll();
        System.out.println(indexNode.value);
        if (indexNode.left != null) {
            queue.add(indexNode.left);
        }
        if (indexNode.right != null) {
            queue.add(indexNode.right);
        }
    }
}
